bryce_zion_birds <- nps |>
  filter(
    ParkName == "Zion National Park" |
    ParkName == "Bryce Canyon National Park",
    CategoryName == "Bird",
    !is.na(Order),
    !is.na(Abundance)
  ) |>
  group_by(ParkCode, Order, CommonNames, Abundance, Nativeness) |>
  summarize(Count = n())
